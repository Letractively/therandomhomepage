28  /***
29   * Simple popup calculator.  The popup calculator is not meant
30   * to be very configurable and does not make it easy to alter
31   * the appearance.  It was designed specifically to allow
32   * for quick use with almost no options.
33   * 
34   * Custom calculators may be created using the SimpleCalcPanel
35   * widget as a base, or perhaps by using the CalcEngine class
36   * directly.
37   * 
38   * A default set of styles are provided when including
39   * the following style-sheet declaration in your HTML
40   * file.
41   * 
42   * <link rel="stylesheet" type="text/css" href="style/gwl-calcPanel.css">
43   *
44   * @author rhanson
45   */
46  public final class PopupCalcPanel extends PopupPanel
47  {
48  
49      /***
50       * Create new calculator popup widget.  The widget will be 
51       * displayed when the widget defined by showCalcButton is 
52       * clicked, and will appear directly below the calcTextBox 
53       * widget.
54       * 
55       * @param calcTextBox
56       * @param showCalcBtn
57       */
58      public PopupCalcPanel (final TextBoxBase calcTextBox, Button showCalcBtn)
59      {
60          super();
61  
62          final SimpleCalcPanel calc = new SimpleCalcPanel();
63          
64          DockPanel calcPanel = new DockPanel();
65          calcPanel.setStyleName("popupCalc-panel");
66          
67          HorizontalPanel actionButtons = new HorizontalPanel();
68          actionButtons.setStyleName("popupCalc-actionButtons");
69          
70          Button doneBtn = new Button("DONE");
71          doneBtn.setStyleName("popupCalc-doneButton");
72          actionButtons.add(doneBtn);
73  
74          Button cancelBtn = new Button("CANCEL");
75          cancelBtn.setStyleName("popupCalc-cancelButton");
76          actionButtons.add(cancelBtn);
77          
78          calcPanel.add(actionButtons, DockPanel.SOUTH);
79          calcPanel.add(calc, DockPanel.CENTER);
80          
81          
82          final PopupPanel popup = this;
83          popup.add(calcPanel);
84          popup.setVisible(false);
85  
86          
87          // "Show Calc" button handler
88          showCalcBtn.addClickListener(new ClickListener()
89          {
90              public void onClick (Widget sender)
91              {
92                  popup.setPopupPosition(calcTextBox.getAbsoluteLeft(),
93                          calcTextBox.getAbsoluteTop() + calcTextBox.getOffsetHeight());
94                  calc.clearValue();
95                  popup.setVisible(true);
96              }
97          });
98  
99          // "Done" button handler
100         doneBtn.addClickListener(new ClickListener()
101         {
102             public void onClick (Widget sender)
103             {
104                 popup.setVisible(false);
105                 calcTextBox.setText(calc.getValue());
106             }
107         });
108 
109         // "Cancel" button handler
110         cancelBtn.addClickListener(new ClickListener()
111         {
112             public void onClick (Widget sender)
113             {
114                 popup.setVisible(false);
115             }
116         });
117     }

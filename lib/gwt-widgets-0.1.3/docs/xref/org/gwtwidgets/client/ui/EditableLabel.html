34  /***
35   * Editable Label class, funcionality displays a Label UI 
36   * Element until clicked on, then if element is set to be 
37   * editable (default) then an editable area and Buttons are 
38   * displayed instead.
39   *
40   * If the Label is not set to be word wrapped (default) then 
41   * the editable area is a Text Box and clicking the OK button 
42   * or hitting return key in the TextBox will display the Label with 
43   * the updated text. 
44   *
45   * If the Label is set to be word wrapped, using the setWordWrap(boolean)
46   * method, then the editable area is a Text Area and clicking the OK 
47   * button will display the Label with the updated text.
48   *
49   * In both cases, clicking Cancel button or hitting Escape key in the 
50   * TextBox/TextArea then the Label is displayed with original text.
51   * 
52   * @author Adam Tacy
53   * @version 0.0.2
54   * 
55   * Changes since version 0.0.1
56   * + made cancelLabelChange public [ref request id: 1518134]
57   * + made originalText have default value of empty string [to support ref request id: 1518134] 
58   * *End*
59   */
60  public class EditableLabel extends Composite implements HasWordWrap, HasText

152     /***
153      * Change the displayed label to be a TextBox and copy label 
154      * text into the TextBox.
155      *
156      */
157     private void changeTextLabel ()
158     {
159         if (isEditable) {
160             // Set up the TextBox
161             originalText = text.getText();
162             
163             // Change the view from Label to TextBox and Buttons
164             text.setVisible(false);
165             confirmChange.setVisible(true);
166             cancelChange.setVisible(true);
167 
168             if (text.getWordWrap()){
169                 // If Label word wrapped use the TextArea to edit
170                 changeTextArea.setText(originalText);
171                 changeTextArea.setVisible(true);
172                 changeTextArea.setFocus(true);
173             } else {
174                 // Otherwise use the TextBox to edit.
175                 changeText.setText(originalText);
176                 changeText.setVisible(true);
177                 changeText.setFocus(true);
178             }
179             
180             //Set instance as being in editing mode.
181             isEditing = true;
182         }
183     }

185     /***
186      * Restores visibility of Label and hides the TextBox and Buttons
187      *
188      */
189     private void restoreVisibility ()
190     {
191         // Change appropriate visibilities
192         text.setVisible(true);
193         confirmChange.setVisible(false);
194         cancelChange.setVisible(false);
195         if(text.getWordWrap()){
196             // If Label is word wrapped hide the TextArea
197             changeTextArea.setVisible(false);
198         } else {
199             // Otherwise hide the TextBox
200             changeText.setVisible(false);
201         }
202         // Set isEditing flag to false as we are no longer editing
203         isEditing = false;
204     }

206     /***
207      * Sets the Label text to the new value, restores the 
208      * display and calls the update method.
209      *
210      */
211     private void setTextLabel ()
212     {
213         if(text.getWordWrap()){
214             // Set the Label to be the text in the Text Box
215             text.setText(changeTextArea.getText());        
216         } else {
217             // Set the Label to be the text in the Text Box
218             text.setText(changeText.getText());
219         }
220         // Set the object back to display label rather than TextBox and Buttons
221         restoreVisibility();
222 
223         // Call the update method provided in the Constructor
224         // (this could be anything from alerting the user through to
225         // Making an AJAX call to store the data.
226         updater.onChange(this);
227     }

229     /***
230      * Sets the Label text to the original value, restores the display.
231      *
232      */
233     public void cancelLabelChange ()
234     {
235         // Set the Label text back to what it was originally
236         text.setText(originalText);         
237         // Set the object back to display Label rather than TextBox and Buttons
238         restoreVisibility();

241     /***
242      * Creates the Label, the TextBox and Buttons.  Also associates
243      * the update method provided in the constructor with this instance.
244      * 
245      * @param labelText The value of the initial Label.
246      * @param onUpdate The class that provides the update method called when the Label has been updated.
247      * @param visibleLength The visible length (width) of the TextBox/TextArea.
248      * @param maxLength The maximum length of text in the TextBox.
249      * @param maxHeight The maximum number of visible lines of the TextArea
250      * @param okButtonText The text diplayed in the OK button.
251      * @param cancelButtonText The text displayed in the Cancel button.
252      */
253     private void createEditableLabel (String labelText, ChangeListener onUpdate,
254             String okButtonText, String cancelButtonText)
255     {
256         // Put everything in a VerticalPanel
257         FlowPanel instance = new FlowPanel();
258 
259         // Create the Label element and add a ClickListener to call out Change method when clicked
260         text = new Label(labelText);
261         text.setStyleName("editableLabel-label");
262 
263         text.addClickListener(new ClickListener()
264         {
265             public void onClick (Widget sender)
266             {
267                 changeTextLabel();
268             }
269         });
270 
271         // Create the TextBox element used for non word wrapped Labels 
272         // and add a KeyboardListener for Return and Esc key presses
273         changeText = new TextBox();
274         changeText.setStyleName("editableLabel-textBox");
275 
276         changeText.addKeyboardListener(new KeyboardListenerAdapter()
277         {
278             public void onKeyPress (Widget sender, char keyCode, int modifiers)
279             {
280                 // If return then save, if Esc cancel the change, otherwise do nothing
281                 switch (keyCode) {
282                     case 13:
283                         setTextLabel();
284                         break;
285                     case 27:
286                         cancelLabelChange();
287                         break;
288                 }
289             }
290         });
291 
292         // Create the TextAre element used for word-wrapped Labels 
293         // and add a KeyboardListener for Esc key presses (not return in this case)
294 
295         changeTextArea = new TextArea();
296         changeTextArea.setStyleName("editableLabel-textArea");
297         
298         changeTextArea.addKeyboardListener(new KeyboardListenerAdapter()
299         {
300             public void onKeyPress (Widget sender, char keyCode, int modifiers)
301             {
302                 // If Esc then cancel the change, otherwise do nothing
303                 switch (keyCode) {
304                     case 27:
305                         cancelLabelChange();
306                         break;
307                 }
308             }
309         });
310         
311         
312         // Set up Confirmation Button
313         confirmChange = createConfirmButton(okButtonText);
314 
315         if (!(confirmChange instanceof SourcesClickEvents)) {
316             throw new RuntimeException("Confirm change button must allow for click events");
317         }
318         
319         ((SourcesClickEvents) confirmChange).addClickListener(new ClickListener()
320         {
321             public void onClick (Widget sender)
322             {
323                 setTextLabel();
324             }
325         });
326 
327         // Set up Cancel Button
328         cancelChange = createCancelButton(cancelButtonText);
329         if (!(cancelChange instanceof SourcesClickEvents)) {
330             throw new RuntimeException("Cancel change button must allow for click events");
331         }
332         
333         ((SourcesClickEvents)cancelChange).addClickListener(new ClickListener()
334         {
335             public void onClick (Widget sender)
336             {
337                 cancelLabelChange();
338             }
339         });
340         
341         // Put the buttons in a panel
342         FlowPanel buttonPanel = new FlowPanel();
343         buttonPanel.setStyleName("editableLabel-buttonPanel");
344         buttonPanel.add(confirmChange);
345         buttonPanel.add(cancelChange);
346         
347         // Add panels/widgets to the widget panel
348         instance.add(text);
349         instance.add(changeText);
350         instance.add(changeTextArea);
351         instance.add(buttonPanel);
352 
353         // Set initial visibilities.  This needs to be after
354         // adding the widgets to the panel because the FlowPanel
355         // will mess them up when added.
356         text.setVisible(true);
357         changeText.setVisible(false);
358         changeTextArea.setVisible(false);
359         confirmChange.setVisible(false);
360         cancelChange.setVisible(false);
361 
362         // Set the updater method.
363         updater = onUpdate;
364 
365         // Assume that this is a non word wrapped Label unless explicitly set otherwise
366         text.setWordWrap(false);
367         
368         // Set the widget that this Composite represents
369         initWidget(instance);
370     }

28  /***
29   * A simple calculator panel.  The panel follows the
30   * template design pattern, which should make it fairly
31   * easy to change the layout and behavior by extending
32   * the class.
33   * 
34   * A default set of styles are provided when including
35   * the following style-sheet declaration in your HTML
36   * file.
37   * 
38   * <link rel="stylesheet" type="text/css" href="style/gwl-calcPanel.css">
39   * 
40   * @author rhanson
41   */
42  public class SimpleCalcPanel extends FlowPanel
43  {
44  
45      private SimpleCalculator calc = null;
46  
47      
48      
49      public SimpleCalcPanel ()
50      {
51          setElement(DOM.createDiv());
52          setStyleName("simpleCalcPanel");
53          init();
54      }
55  
56      
57      protected void init ()
58      {
59          TextBox textDisplay = createTextDisplay();
60          createCalc(createDisplayListener(textDisplay), textDisplay);
61          setKeyboardListener(textDisplay);
62          initComplete();
63      }
64  
65      
66      protected CalcDisplayListener createDisplayListener (final TextBox textDisplay)
67      {
68          return new CalcDisplayListener()
69          {
70              public void onDisplayRefresh (String value)
71              {
72                  textDisplay.setText(value);
73              }
74          };
75      }
76  
77  
78      protected void setKeyboardListener (TextBox textDisplay)
79      {
80          textDisplay.addKeyboardListener(new CalcKeyboardListener(calc, textDisplay));
81      }
82  
83  
84      protected void initComplete ()
85      {
86      }
87  
88  
89      protected void createCalc (CalcDisplayListener cdl, TextBox textDisplay)
90      {
91          calc = new SimpleCalculator(this, cdl, textDisplay);
92      }
93  
94  
95      protected TextBox createTextDisplay ()
96      {
97          TextBox result = new TextBox();
98          result.setStyleName("simpleCalcDisplay");
99          return result;
100     }
101 
102 
103 
104     protected void buildCalcLayout (CalcEngine calcEngine, TextBox textDisplay)
105     {
106         add(textDisplay);
107 
108         HorizontalPanel numPad1 = new HorizontalPanel();
109         numPad1.add(calcEngine.createNumberButton("calcButton-1", "1", 1));
110         numPad1.add(calcEngine.createNumberButton("calcButton-2", "2", 2));
111         numPad1.add(calcEngine.createNumberButton("calcButton-3", "3", 3));
112 
113         HorizontalPanel numPad2 = new HorizontalPanel();
114         numPad2.add(calcEngine.createNumberButton("calcButton-4", "4", 4));
115         numPad2.add(calcEngine.createNumberButton("calcButton-5", "5", 5));
116         numPad2.add(calcEngine.createNumberButton("calcButton-6", "6", 6));
117 
118         HorizontalPanel numPad3 = new HorizontalPanel();
119         numPad3.add(calcEngine.createNumberButton("calcButton-7", "7", 7));
120         numPad3.add(calcEngine.createNumberButton("calcButton-8", "8", 8));
121         numPad3.add(calcEngine.createNumberButton("calcButton-9", "9", 9));
122 
123         HorizontalPanel numPad4 = new HorizontalPanel();
124         numPad4.add(calcEngine.createNumberButton("calcButton-0", "0", 0));
125         numPad4.add(calcEngine.createOpButton("calcButton-negate", "+/-", CalcEngine.OP_NEGATE));
126         numPad4.add(calcEngine.createOpButton("calcButton-decimal", ".", CalcEngine.OP_DEC));
127 
128         VerticalPanel numPad = new VerticalPanel();
129         numPad.add(numPad1);
130         numPad.add(numPad2);
131         numPad.add(numPad3);
132         numPad.add(numPad4);
133 
134         
135         VerticalPanel opPad = new VerticalPanel();
136         opPad.add(calcEngine.createOpButton("calcButton-plus", "+", CalcEngine.OP_PLUS));
137         opPad.add(calcEngine.createOpButton("calcButton-minus", "-", CalcEngine.OP_MINUS));
138         opPad.add(calcEngine.createOpButton("calcButton-multiply", "*", CalcEngine.OP_MULT));
139         opPad.add(calcEngine.createOpButton("calcButton-divide", "/", CalcEngine.OP_DIV));
140 
141         VerticalPanel funcPad = new VerticalPanel();
142         funcPad.add(calcEngine.createOpButton("calcButton-clear", "C", CalcEngine.OP_CLEAR));
143         funcPad.add(calcEngine.createOpButton("calcButton-erase", "CE", CalcEngine.OP_ERASE));
144         funcPad.add(calcEngine.createOpButton("calcButton-equals", "=", CalcEngine.OP_EQUALS));
145 
146         VerticalPanel spacerPad = new VerticalPanel();
147         spacerPad.add(new HTML("&nbsp;"));
148         
149         this.add(numPad);
150         this.add(spacerPad);
151         this.add(opPad);
152         this.add(funcPad);
153     }
154 
155     
156     public String getValue ()
157     {
158         return Double.toString(calc.getTotal());
159     }
160     
161     
162     class SimpleCalculator extends CalcEngine
163     {
164         
165         SimpleCalculator (SimpleCalcPanel calcPanel, CalcDisplayListener cdl, TextBox textDisplay)
166         {
167             super(cdl);
168             calcPanel.buildCalcLayout(this, textDisplay);
169         }
170 
171     }
172 
173 
174 
175     public void clearValue ()
176     {
177         calc.pressOperator(CalcEngine.OP_CLEAR);
178     }

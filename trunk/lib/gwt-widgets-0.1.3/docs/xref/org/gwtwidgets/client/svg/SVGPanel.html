19  import org.gwtwidgets.client.ext.Namespace;

22  public class SVGPanel extends SVGPanelBase
23  {
24      private Namespace ns;
25      private SVGDefs defs;
26      private int genId = 0;
27  
28      
29      /***
30       * Create a new SVG panel of the specified size using
31       * the default namespace and URI.  The default namespace
32       * prefix is "svg", and the URI is "http://www.w3.org/2000/svg".
33       * 
34       * @param width in pixels
35       * @param height in pixels
36       */
37      public SVGPanel (double width, double height)
38      {
39          this(new Namespace("svg", "http://www.w3.org/2000/svg"), width, height);
40      }
41  
42      /***
43       * Create a new SVG panel of the specified size using
44       * a custom namespace.  In most cases you should use
45       * the default namespace, but for cases where you can't
46       * you may specify a custom one with this constructor.
47       * 
48       * @param ns namespace
49       * @param width in pixels
50       * @param height in pixels
51       */
52      public SVGPanel (Namespace ns, double width, double height)
53      {
54          this.ns = ns;
55          setElement(createNsElement(ns, "svg"));
56          setAttributeNS(getElement(), "width", width + "px");
57          setAttributeNS(getElement(), "height", height + "px");
58          
59          this.defs = new SVGDefs(ns);
60          this.add(defs);
61      }
62  
63      /***
64       * Create a new cicle.
65       * 
66       * @param cx x-coordinate for the center of the circle
67       * @param cy y-coordinate for the center of the circle
68       * @param radius of the circle in pixels
69       * @return circle widget
70       */
71      public SVGCircle createCircle (double cx, double cy, double radius)
72      {
73          return new SVGCircle(ns, cx, cy, radius);
74      }

82       * @param rx horizontal radius of the ellipse in pixels
83       * @param ry vertical radius of the ellipse in pixels
84       * @return ellipse widget
85       */
86      public SVGEllipse createEllipse (double cx, double cy, double rx, double ry)

88          return new SVGEllipse(ns, cx, cy, rx, ry);

92      /***

106     public SVGPath createPath (double x, double y)

110     
111     public SVGLinearGradient createLinearGradient ()
112     {
113         String id = getNewId();
114         SVGLinearGradient result = new SVGLinearGradient(ns, id);
115         defs.add(result.getWidget());
116         return result;
117     }

119     public SVGLinearGradient createLinearGradient (String id)
120     {

219     private void fillCalendar ()
220     {
221         clear();
222 
223         for (int row = 0; row < getRowCount(); ++row) {
224             for (int col = 0; col < getCellCount(row); ++col) {
225                 HTML contents = new HTML("&nbsp;");
226                 contents.setStyleName("calendar-cell");
227                 contents.addStyleName(getStylenameForDay(col + firstDayOffset));
228                 setWidget(row, col, contents);
229             }
230         }
231 
232         for (int col = 0 + firstDayOffset; col < (7 + firstDayOffset); col++) {
233             int tCol = col < 7 ? col : col - 7;
234             HTML contents = new HTML(weekDayNames[tCol]);
235             contents.setStyleName("calendar-cell");
236             contents.addStyleName("calendar-headingCell");
237             contents.addStyleName(getStylenameForDay(tCol));
238             setWidget(0, col - firstDayOffset, contents);
239         }
240         
241         int firstDay = cm.getFirstDay() - firstDayOffset;
242         firstDay = firstDay < 0 ? firstDay + 7 : firstDay;
243         
244         for (int d = 0; d < cm.getDays(); d++) {
245             final int date = d + 1;
246             final boolean hasEvents = cm.hasEvent(date);
247             int row = (int) Math.floor((firstDay + d + 6) / 7) + (firstDay == 0 ? 1 : 0);
248             int col = (firstDay + d + 6) % 7;
249             HTML contents = new HTML(Integer.toString(date));
250             contents.setStyleName("calendar-cell");
251             contents.addStyleName("calendar-dateCell");
252             contents.addStyleName(getStylenameForDay(col + firstDayOffset));
253             if (hasEvents) {
254                 contents.addStyleName("calendar-eventCell");
255             }
256             contents.addClickListener(new ClickListener()
257             {
258                 public void onClick (Widget sender)
259                 {
260                     Date dateObj = new Date(cm.getYear() - 1900, cm.getMonth(), date, 0, 0, 0);
261                     CalendarDate calDate = new CalendarDate(dateObj, cm.getEvents(date));
262 
263                     if (hasEvents) {
264                         factory.fireEventDateClick(calDate);
265                     }
266                     else {
267                         factory.fireDateClick(calDate);
268                     }
269                 }
270             });
271             setWidget(row, col, contents);
272         }
273     }

275     private String getStylenameForDay (int col)
276     {
277         if (col > 6) {
278             col -= 7;
279         };
280         
281         switch (col) {
282             case 0:
283                 return "calendar-cellSunday";
284             case 1:
285                 return "calendar-cellMonday";
286             case 2:
287                 return "calendar-cellTuesday";
288             case 3:
289                 return "calendar-cellWednesday";
290             case 4:
291                 return "calendar-cellThursday";
292             case 5:
293                 return "calendar-cellFriday";
294             case 6:
295                 return "calendar-cellSaturday";
296             default:
297                 break;
298         }
299         return "";
300     }

437     /***
438      * Set the number of columns to shift calendar days. Use
439      * this to change what the first day of the week is.
440      * Valid values are 0-6.  Values outside of this range
441      * will throw a RuntimeException.
442      * 
443      * The following will make Monday the first day of the week,
444      * causing the calendar headings to read M, T, W, T, F, S, S.
445      * 
446      * cal.setFirstDayOffset(CalendarPanel.OFFSET_MONDAY);
447      * 
448      * @param firstDayOffset offset in days
449      */
450     public void setFirstDayOffset (int firstDayOffset)
451     {
452         if (firstDayOffset < 0 || firstDayOffset > 6) {
453             throw new RuntimeException("offset out of bounds");
454         }
455
